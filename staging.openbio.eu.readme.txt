### SETTING EMAIL SERVER ###
Follow firections from: https://www.digitalocean.com/community/tutorials/how-to-install-and-configure-postfix-on-ubuntu-16-04
sudo DEBIAN_PRIORITY=low apt-get install postfix

sudo postconf -e 'home_mailbox= Maildir/'
sudo postconf -e 'virtual_alias_maps= hash:/etc/postfix/virtual'
sudo vim /etc/postfix/virtual

### CREATE VIRTUAL EMAILS
cat /etc/postfix/virtual
contact@staging.openbio.eu alexandros.kanterakis@gmail.com

sudo postmap /etc/postfix/virtual
sudo systemctl restart postfix

echo 'export MAIL=~/Maildir' | sudo tee -a /etc/bash.bashrc | sudo tee -a /etc/profile.d/mail.sh
